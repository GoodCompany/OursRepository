CREATE VIEW [dbo].[V_SearchStock]
AS
SELECT     dbo.MiU_Kabinet.kab_Nkab, dbo.MiU_Kabinet.kab_Tipkab, dbo.MiU_Product.Prd_code, dbo.MiU_Product.Prd_name, dbo.MiU_Stock.stk_Quantity, 
                      dbo.MiU_Product.prd_dlina * dbo.MiU_Product.prd_shir * dbo.MiU_Product.prd_visota AS Product_OB, dbo.MiU_Product.prd_ves, 
                      dbo.MiU_Product.prd_cena, dbo.MiU_Product.prd_cena * dbo.MiU_Stock.stk_Quantity AS sumcena, CONVERT(varchar(10), dbo.MiU_Stock.stk_dateIN, 
                      104) AS dataIN, dbo.MiU_Product.prd_armor, DATEDIFF(month, dbo.MiU_Stock.stk_dateIN, GETDATE()) 
                      * dbo.MiU_Product.prd_armor AS tekushiy_armor, ISNULL(dbo.MiU_Dogovor.dg_name, '-') AS Ndogovor, ISNULL(dbo.MiU_Company.cmp_Name, '-') 
                      AS postavshik
FROM         dbo.MiU_Stock WITH (nolock) INNER JOIN
                      dbo.MiU_Kabinet WITH (nolock) ON dbo.MiU_Stock.stk_kabID = dbo.MiU_Kabinet.kab_ID INNER JOIN
                      dbo.MiU_Product WITH (nolock) ON dbo.MiU_Stock.stk_prdID = dbo.MiU_Product.Prd_ID LEFT OUTER JOIN
                      dbo.MiU_Dogovor WITH (nolock) ON dbo.MiU_Stock.stk_dgID = dbo.MiU_Dogovor.dg_ID LEFT OUTER JOIN
                      dbo.MiU_Company WITH (nolock) ON dbo.MiU_Company.cmp_ID = dbo.MiU_Stock.stk_cmpID

GO
